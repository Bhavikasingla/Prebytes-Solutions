#include <bits/stdc++.h>
  using namespace std;
  
  int main()
  {
    //write your code here
    
    int t;
    cin>>t;
    
    while(t--){
      
      int n;
      cin>>n;
      
      int arr[n];
      for(int i=0;i<n;i++){
        cin>>arr[i];
      }

      int sum=0;
      int ans=INT_MIN;

      for(int i=0;i<n;i++)
      { 
        sum+=arr[i];
        if(sum > ans){
          ans = sum;
        }
        if(sum < 0){
          sum=0;
        }  
      }
      cout<<ans<<endl;
    }
    return 0;
  }
