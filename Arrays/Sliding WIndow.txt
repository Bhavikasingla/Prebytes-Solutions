#include <bits/stdc++.h>
  using namespace std;
  
  int main()
  {
    
    int t;
    cin>>t;
    
    while(t--)
    
    {
      
      int n,k;
      
      cin>>n>>k;
    
      int arr[n];
    
      for(int i=0;i<n;i++)
      {
        cin>>arr[i];
      }
    
      int max=INT_MIN;
      int sum=0;
    
      for(int i=0;i<k;i++)
      {
        sum+=arr[i];
      }
    
      max=sum;
    
      for(int i=k;i<n;i++)
      {
        sum=sum+arr[i]-arr[i-k];
    
        if(sum>max)
        {
          max=sum;
        }
    
      }
    
      cout<<max<<endl;

    }
  
    
    return 0;
  }
