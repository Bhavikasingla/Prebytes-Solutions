Q8 Which of the following is the correct output for the program given below?
int main() {
  int i = 4, j = -1, k = 0, w, x, y, z;
  w = i || j || k;
  x = i && j && k;
  y = i || j && k;
  z = i && j || k;
  cout<< w <<" "<< x <<" "<< y <<" "<< z;
  return 0;
}
Ans 1 0 1 1
